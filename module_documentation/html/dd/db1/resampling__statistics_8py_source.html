22 def randomization(statfunction, fulldata, testdata, iteration=1000):
23  """
24  Test harness for statistical randomization test, also known as
25  permutation test or bootstrapping.
26 
27  Pitman, EJG. 1938. Significance tests which may be applied to samples
28  from any population. Part III. The analysis of variance test.
29  Biometrika 29: 322-335.
30  Manly, Bryan FJ. 2007. Randomization, Bootstrap and Monte Carlo
31  Methods in Biology (3rd edition). Chapman & Hall/CRC.
32 
33  Parameters:
34  statfunction = Python function to calculate the test statistic.
35  fulldata = list of entire data set (for randomization)
36  testdata = list of sample data of interest (for testing)
37  iteration = number of times random samples are drawn.
38  Default = 1000
39 
40  Returns:
41  Proportion where statistic from randomized sample is greater
42  or equal to the statistic from testdata. This can be taken as
43  p-value for the test
44  """
45  from copads.Operations import sample_wr
46  test_statistic = statfunction(testdata)
47  randomized_statistic = [statfunction(sample_wr(fulldata, len(testdata)))
48  for i in range(iteration)]
49  #print randomized_statistic, test_statistic
50  more = [1 for x in randomized_statistic
51  if x >= test_statistic]
52  return sum(more) / float(len(randomized_statistic))

54 def sample_bootstrap(data, iteration=1000):
55  """
56  Calculates bootstrapped means and standard deviation.
57 
58  Efron, Bradley. 1979. Bootstrap methods: Another look at the jackknife.
59  The Annals of Statistics 7:1-26.
60  Manly, Bryan FJ. 2007. Randomization, Bootstrap and Monte Carlo
61  Methods in Biology (3rd edition). Chapman & Hall/CRC.
62 
63  Parameters:
64  data = list of data to generate bootstrapped samples from.
65  iterations = number of times bootstrapped samples are drawn.
66  Default = 1000
67 
68  Returns:
69  (bootstrapped means, bootstrapped standard deviation)
70  """
71  from copads.Operations import sample_wr
72  len_data = len(data)
73  pool = [sum(sample_wr(data, len_data)) / float(len_data)
74  for x in range(iteration)]
75  poolmeans = float(sum(pool)) / float(len(pool))
76  div = [(x - poolmeans)**2 for x in pool]
77  stdev = math.sqrt(sum(div)) / (len(pool) - 1)
78  return (poolmeans, stdev)

80 def sample_jackknife(data):
81  """
82  Estimates mean and standard deviation from a single sample
83  using Jackknife method.
84 
85  Wu, CFJ. 1986. Jackknife, Bootstrap and other resampling methods
86  in regression analysis. The Annals of Statistics 14:1261–1295.
87 
88  Parameters:
89  data = list of data to generate bootstrapped samples from.
90 
91  Returns:
92  (estimated means, estimated standard deviation)
93  """
94  len_data = len(data)
95  pool = [sum(data.pop(i)) / float(len_data -1)
96  for i in range(len_data)]
97  poolmeans = float(sum(pool)) / float(len(pool))
98  div = [(x - poolmeans)**2 for x in pool]
99  stdev = math.sqrt(sum(div)) / (len(pool) - 1)
100  return (poolmeans, stdev)
